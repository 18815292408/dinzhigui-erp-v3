用以下逻辑替换现有 getCustomers 函数，同时考虑 orders 和 designs 两个表：

```typescript
async function getCustomers() {
  const cookieStore = await cookies()
  const sessionCookie = cookieStore.get('session')

  if (!sessionCookie) {
    return { withoutOrders: [], withOrders: [] }
  }

  const user = parseSessionUser(sessionCookie.value)
  if (!user) {
    return { withoutOrders: [], withOrders: [] }
  }

  const adminSupabase = await createAdminClient()

  // 获取所有客户
  const { data: allCustomers } = await adminSupabase
    .from('customers')
    .select('*')
    .eq('organization_id', user.organization_id)
    .order('created_at', { ascending: false })

  // 获取所有订单
  const { data: allOrders } = await adminSupabase
    .from('orders')
    .select('customer_name, status, order_no, signed_amount')
    .eq('organization_id', user.organization_id)

  // 获取有进行中设计（draft, submitted）的客户ID
  const { data: activeDesigns } = await adminSupabase
    .from('designs')
    .select('customer_id, status')
    .eq('organization_id', user.organization_id)
    .in('status', ['draft', 'submitted'])

  // 计算有进行中订单的客户名称集合
  const customerNamesWithActiveOrders = new Set(
    (allOrders || [])
      .filter(o => o.status !== 'completed')
      .map(o => o.customer_name)
      .filter(Boolean)
  )

  // 计算有进行中设计的客户ID集合
  const customerIdsWithActiveDesigns = new Set(
    (activeDesigns || [])
      .map(d => d.customer_id)
      .filter(Boolean)
  )

  // 判断客户是否在"订单跟进"
  const isInFollowup = (c: any) =>
    customerNamesWithActiveOrders.has(c.name) ||
    customerIdsWithActiveDesigns.has(c.id)

  // 分类客户
  const customersWithOrders: any[] = []
  const customersWithoutOrders: any[] = []

  for (const c of allCustomers || []) {
    if (isInFollowup(c)) {
      // 附加订单和设计信息
      const orders = (allOrders || []).filter(
        o => o.customer_name === c.name && o.status !== 'completed'
      )
      const designs = (activeDesigns || []).filter(d => d.customer_id === c.id)
      customersWithOrders.push({ ...c, orders, designs })
    } else {
      customersWithoutOrders.push(c)
    }
  }

  return {
    withoutOrders: customersWithoutOrders,
    withOrders: customersWithOrders
  }
}
```
